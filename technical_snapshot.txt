import React, { useState } from 'react';
import { useStore } from '../store/useStore';
import { Send } from 'lucide-react';
import type { CommunityPost } from '../types';

interface CreatePostFormProps {
    activeChannel: string;
}

export const CreatePostForm: React.FC<CreatePostFormProps> = ({ activeChannel }) => {
    const [title, setTitle] = useState('');
    const [content, setContent] = useState('');
    const { addPost } = useStore();
    const MAX_CHARS = 500;

    const handleSubmit = (e: React.FormEvent) => {
        e.preventDefault();
        if (!title.trim() || !content.trim()) return;

        const newPost: CommunityPost = {
            id: Date.now(),
            userId: 1, // Hardcoded for the current user (Napoleón)
            channel: activeChannel,
            title: title,
            content: content,
            likes: 0,
            comments: 0,
            timestamp: 'ahora mismo'
        };

        addPost(newPost);
        setTitle('');
        setContent('');
    };

    const handleContentChange = (e: React.ChangeEvent<HTMLTextAreaElement>) => {
        if (e.target.value.length <= MAX_CHARS) {
            setContent(e.target.value);
        }
    };

    const charsRemaining = MAX_CHARS - content.length;

    return (
        <form onSubmit={handleSubmit} className="bg-white p-6 rounded-xl shadow-lg border border-gray-200 mb-8">
            <h3 className="text-xl font-bold text-gray-800 mb-4">Crear nueva publicación en <span className="text-purple-600">{activeChannel}</span></h3>
            <input
                type="text"
                className="w-full p-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-purple-500 focus:border-transparent text-base text-gray-700 placeholder-gray-400 transition-all duration-200 mb-4"
                placeholder="Título de la publicación"
                value={title}
                onChange={(e) => setTitle(e.target.value)}
            />
            <textarea
                className="w-full p-3 border border-gray-300 rounded-lg focus:ring-2 focus:ring-purple-500 focus:border-transparent resize-y text-base text-gray-700 placeholder-gray-400 transition-all duration-200"
                rows={4}
                placeholder="Contenido de la publicación..."
                value={content}
                onChange={handleContentChange}
            ></textarea>
            <div className="flex justify-between items-center mt-3">
                <span className={`text-sm ${charsRemaining < 50 ? 'text-red-500' : 'text-gray-500'}`}>
                    {charsRemaining} caracteres restantes
                </span>
                <button
                    type="submit"
                    className="bg-gradient-to-r from-purple-600 to-blue-500 text-white font-semibold py-2 px-6 rounded-lg flex items-center gap-2 transition-all duration-200 ease-in-out shadow-md hover:shadow-lg hover:from-purple-700 hover:to-blue-600 disabled:opacity-50 disabled:cursor-not-allowed"
                    disabled={!title.trim() || !content.trim() || content.length > MAX_CHARS}
                >
                    <Send className="w-5 h-5" /> Publicar
                </button>
            </div>
        </form>
    );
};

import React, { useState, useEffect } from 'react';
import { useStore } from '../store/useStore';
import { X, Copy, Check } from 'lucide-react';

export const PromptModal: React.FC = () => {
    const { isModalOpen, selectedPrompt, closeModal } = useStore();
    const [isCopied, setIsCopied] = useState(false);

    useEffect(() => {
        if (isModalOpen) {
            setIsCopied(false);
        }
    }, [isModalOpen]);

    if (!isModalOpen || !selectedPrompt) {
        return null;
    }

    const handleCopy = () => {
        const textArea = document.createElement("textarea");
        textArea.value = selectedPrompt.promptText;
        textArea.style.position = "fixed";
        textArea.style.top = "-9999px";
        textArea.style.left = "-9999px";
        document.body.appendChild(textArea);
        textArea.select();
        try {
            document.execCommand('copy');
            setIsCopied(true);
            setTimeout(() => setIsCopied(false), 2000);
        } catch (err) {
            console.error('Fallback: Oops, unable to copy', err);
        }
        document.body.removeChild(textArea);
    };

    return (
        <div className="fixed inset-0 bg-black bg-opacity-60 backdrop-blur-sm flex justify-center items-center z-50 animate-fade-in-fast" onClick={closeModal}>
            <div className="bg-white rounded-xl shadow-2xl w-11/12 max-w-2xl max-h-[90vh] flex flex-col" onClick={(e) => e.stopPropagation()}>
                <header className="p-6 border-b border-gray-200 flex justify-between items-center">
                    <h2 className="text-2xl font-bold text-gray-800">{selectedPrompt.title}</h2>
                    <button onClick={closeModal} className="text-gray-400 hover:text-gray-600"><X className="w-6 h-6" /></button>
                </header>
                <div className="p-6 overflow-y-auto"><pre className="bg-gray-50 p-4 rounded-lg text-sm text-gray-700 whitespace-pre-wrap font-mono">{selectedPrompt.promptText}</pre></div>
                <footer className="p-6 border-t border-gray-200">
                    <button onClick={handleCopy} className={`w-full flex items-center justify-center gap-2 font-bold py-3 px-6 rounded-lg transition-colors ${isCopied ? 'bg-green-500 text-white' : 'bg-gray-800 text-white hover:bg-gray-900'}`}>
                        {isCopied ? <Check className="w-5 h-5" /> : <Copy className="w-5 h-5" />}
                        {isCopied ? '¡Copiado!' : 'Copiar Prompt'}
                    </button>
                </footer>
            </div>
        </div>
    );
};

import { create } from 'zustand';
import type { Prompt, CommunityUser, CommunityPost } from '../types';
import { promptsData } from '../data/prompts';
import { initialCommunityData } from '../data/community';

export type View = 'inicio' | 'synta-lab' | 'arsenal' | 'comunidad' | 'aloha-academy' | 'build-ecosystem' | 'profile';

interface AppState {
  currentView: View;
  setCurrentView: (view: View) => void;
  isLoggedIn: boolean;
  userName: string;
  login: () => void;
  logout: () => void;
  isSidebarOpen: boolean;
  toggleSidebar: () => void;
  prompts: Prompt[];
  searchTerm: string;
  activeCategory: string;
  setSearchTerm: (term: string) => void;
  setActiveCategory: (category: string) => void;
  isModalOpen: boolean;
  selectedPrompt: Prompt | null;
  openModal: (prompt: Prompt) => void;
  closeModal: () => void;
  communityUsers: CommunityUser[];
  communityPosts: CommunityPost[];
  addPost: (post: CommunityPost) => void;
  toggleLikePost: (postId: number) => void;
  updateProfile: (updatedUser: Partial<CommunityUser>) => void;
}

export const useStore = create<AppState>((set) => ({
  currentView: 'inicio',
  setCurrentView: (view) => set({ currentView: view }),
  isLoggedIn: false,
  userName: 'Napoleón',
  login: () => set({ isLoggedIn: true }),
  logout: () => set({ isLoggedIn: false, currentView: 'inicio' }),
  isSidebarOpen: false,
  toggleSidebar: () => set((state) => ({ isSidebarOpen: !state.isSidebarOpen })),
  prompts: promptsData,
  searchTerm: '',
  activeCategory: 'Todos',
  setSearchTerm: (term) => set({ searchTerm: term }),
  setActiveCategory: (category) => set({ activeCategory: category }),
  isModalOpen: false,
  selectedPrompt: null,
  openModal: (prompt) => set({ isModalOpen: true, selectedPrompt: prompt }),
  closeModal: () => set({ isModalOpen: false, selectedPrompt: null }),
  communityUsers: initialCommunityData.users,
  communityPosts: initialCommunityData.posts,
  addPost: (post) => set((state) => ({ communityPosts: [post, ...state.communityPosts] })),
  toggleLikePost: (postId) => set((state) => ({
    communityPosts: state.communityPosts.map(post =>
      post.id === postId ? { ...post, likes: post.likes + 1 } : post
    ),
  })),
  updateProfile: (updatedUser) => set((state) => ({
    communityUsers: state.communityUsers.map(user =>
      user.id === updatedUser.id ? { ...user, ...updatedUser } : user
    ),
  })),
}));

import { useStore } from '../store/useStore';

import type { CommunityUser } from '../types';

export const ProfileView: React.FC = () => {
    const { userName, communityUsers, communityPosts, updateProfile } = useStore();
    const currentUser = communityUsers.find(user => user.name === userName);

    const [isEditing, setIsEditing] = useState(false);
    const [editedName, setEditedName] = useState(currentUser?.name || '');
    const [editedEmail, setEditedEmail] = useState(currentUser?.email || '');
    const [editedAvatar, setEditedAvatar] = useState(currentUser?.avatar || '');
    const [editedTitle, setEditedTitle] = useState(currentUser?.title || '');

    if (!currentUser) {
        return <div className="p-6 text-center text-gray-600">Usuario no encontrado.</div>;
    }

    const handleSave = () => {
        if (currentUser) {
            updateProfile({
                id: currentUser.id,
                name: editedName,
                email: editedEmail,
                avatar: editedAvatar,
                title: editedTitle,
            } as Partial<CommunityUser>);
        }
        setIsEditing(false);
    };

    const userPosts = communityPosts.filter(post => post.userId === currentUser.id);

    return (
        <div className="p-6 md:p-10 animate-fade-in h-full bg-gray-50">
            <div className="max-w-4xl mx-auto bg-white rounded-xl shadow-lg border border-gray-200 p-8">
                <h1 className="text-3xl md:text-4xl font-bold text-gray-900 mb-6 text-center">Perfil de Usuario</h1>

                <div className="flex flex-col items-center mb-8">
                    <div className="relative">

                        {isEditing && (

                                <span className="text-sm text-gray-700 font-semibold">Nivel: {currentUser.level}</span>

                            </button>
                        )}
                        {isEditing && (
                            <button onClick={() => setIsEditing(false)} className="bg-gray-300 text-gray-800 px-6 py-2 rounded-lg font-semibold hover:bg-gray-400 transition-colors">
                                Cancelar
                            </button>
                        )}
                    </div>
                </div>

                <div className="mt-10 border-t border-gray-200 pt-8">
                    <h3 className="text-2xl font-bold text-gray-800 mb-6">Mis Publicaciones Recientes</h3>
                    {userPosts.length > 0 ? (

                                    <span className="text-xs text-gray-400 mt-2 block">{post.timestamp}</span>
                                </div>
                            ))}
                        </div>
                    ) : (
                        <p className="text-gray-500 text-center">Aún no has realizado ninguna publicación.</p>

                        <li><a href="#" className="text-blue-600 hover:underline">Configuración de Notificaciones</a></li>
                    </ul>
                </div>

        </div>
    );
};

import React, { useMemo } from 'react';
import { useStore } from '../store/useStore';
import { PromptCard } from '../components/PromptCard';
import { Search } from 'lucide-react';

export const SyntaLabView: React.FC = () => {
    const { prompts, searchTerm, setSearchTerm, activeCategory, setActiveCategory } = useStore();
    
    const categories = useMemo(() => ['Todos', ...new Set(prompts.map(p => p.category))], [prompts]);
    
    const filteredPrompts = useMemo(() => {
        return prompts.filter(prompt => {
            const matchesCategory = activeCategory === 'Todos' || prompt.category === activeCategory;
            const matchesSearch = searchTerm === '' || 
                                  prompt.title.toLowerCase().includes(searchTerm.toLowerCase()) || 
                                  prompt.description.toLowerCase().includes(searchTerm.toLowerCase()) || 
                                  prompt.tags.some(tag => tag.toLowerCase().includes(searchTerm.toLowerCase()));
            return matchesCategory && matchesSearch;
        });
    }, [prompts, searchTerm, activeCategory]);

    return (
        <div className="p-6 md:p-10 animate-fade-in">
            <h1 className="text-3xl md:text-4xl font-bold text-gray-800 mb-2">Laboratorio de Prompts</h1>
            <p className="text-gray-500 text-lg mb-8">Un arsenal de herramientas para construir pensamiento estratégico. Busca, filtra y encuentra el prompt preciso para tu objetivo.</p>
            
            <div className="sticky top-0 md:top-16 bg-gray-50/80 backdrop-blur-sm py-4 z-10">
                <div className="relative mb-4">
                    <Search className="absolute left-4 top-1/2 -translate-y-1/2 text-gray-400 w-5 h-5" />
                    <input 
                        type="text" 
                        placeholder="Buscar por título, descripción o etiqueta..." 
                        value={searchTerm} 
                        onChange={(e) => setSearchTerm(e.target.value)} 
                        className="w-full pl-12 pr-4 py-3 bg-white border border-gray-300 rounded-full focus:outline-none focus:ring-2 focus:ring-purple-500" 
                    />
                </div>
                <div className="flex flex-wrap gap-2">
                    {categories.map(category => (
                        <button 
                            key={category} 
                            onClick={() => setActiveCategory(category)} 
                            className={`px-4 py-2 text-sm font-semibold rounded-full transition-colors ${activeCategory === category ? 'bg-gray-800 text-white' : 'bg-white text-gray-600 hover:bg-gray-200'}`}
                        >
                            {category}
                        </button>
                    ))}
                </div>
            </div>

            <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6 mt-8">
                {filteredPrompts.length > 0 ? (
                    filteredPrompts.map((prompt, index) => <PromptCard key={index} prompt={prompt} />)
                ) : (
                    <p className="md:col-span-3 text-center text-gray-500">No se encontraron prompts que coincidan con tu búsqueda.</p>
                )}
            </div>
        </div>
    );
};
